#!/system/bin/sh

# Đảm bảo script có quyền root
if [ "$(id -u)" != "0" ]; then
  echo "This script must be run as root" 1>&2
  exit 1
fi

# Đặt tốc độ tối đa của CPU
echo "1.8 GHz" > /sys/devices/system/cpu/cpu0/cpufreq/scaling_max_freq

# Đặt tốc độ tối thiểu của CPU (nếu cần)
echo "1.2 GHz" > /sys/devices/system/cpu/cpu0/cpufreq/scaling_min_freq

echo "CPU frequency settings applied."
